function ProvinceStats() {
  const [stats, setStats] = useState(null);

    const s = getProvinceStats();
    setStats(s);
  }, []);

  return (
    <div>
      <p>Total provinces: {stats.totalProvinces}</p>
      <p>Total communes: {stats.totalCommunes}</p>
      <p>Largest: {stats.largestProvince.name}</p>
    </div>
  );
}

üîç Drilldown Example

import { useState, useEffect } from 'react';
import { loadProvinceCommunes } from '@xdev-asia/vietnam-map-34-provinces/core';

function ProvinceDetail({ provinceName }) {
  const [communes, setCommunes] = useState([]);
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    if (!provinceName) return;
    
    setLoading(true);
    loadProvinceCommunes(provinceName)
      .then(data => {
        setCommunes(data?.features || []);
      })
      .finally(() => setLoading(false));
  }, [provinceName]);

  if (loading) return <div>Loading communes...</div>;

  return (
    <div>
      <h3>{provinceName} - {communes.length} x√£/ph∆∞·ªùng</h3>
      <ul>
        {communes.slice(0, 10).map(c => (
          <li key={c.properties.GID_3}>
            {c.properties.NAME_3} ({c.properties.TYPE_3})
          </li>
        ))}
      </ul>
    </div>
  );
}
